  Display following details for each student:
        
        -   Name

        -   Standard

        -   Sex
Ans
select Student_3.Name, Studentdetail_3.standard, Studentdetail_3.flagsex  from Student_3 inner join
Studentdetail_3 on Student_3.recid=Studentdetail_3.recid

    -   Find SUM of student hours.			Fields : Name, TotalHours
Ans
select Student_3.Name , sum(Studenthours_3.hours) from Student_3 inner join Studenthours_3
 on Student_3.recid=Studenthours_3.recid group by Student_3.recid
***************************************************************************************

    -   Find Average Present of each student.		Fields : Name, Present
Ans
select Student_3.Name , avg(Studentpresent_3.present) from Student_3 inner join Studentpresent_3
 on Student_3.recid=Studentpresent_3.recid group by Student_3.recid
