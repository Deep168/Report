Handling Events with class Component

Profile.js

import React from 'react';
import "./style.css";
 class Profile extends React.Component{

  test(){
    alert("Test function called")
  }
   render(){
     return (
     <div>
       <h1> Profile </h1>
       <button onClick={this.test.bind()}> Profile </button>
     </div>
     )
   }
 }

 export default Profile

App.js

import React from "react";

 export default function App() {
  return (

}
